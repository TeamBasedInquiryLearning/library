\item Given the equation, $3x-4y=8$, solve for $y$ in terms of $x$.

  \begin{enumerate}
  \begin{multicols}{4}
  \item $y=8-3x$
  \item $y=8-\dfrac{3x}{4}$
  \item $y=-3x-2$
  \item $y=\dfrac{3}{4}x-2$ %correct
   \end{multicols}
  \end{enumerate}

  \item $x=(y+3)^{2}+5$
  \item $x=(3-y)^{2}-5$ %correct
  \item $x=(y-3)^2+5$ 
   \end{multicols}
  \end{enumerate}
