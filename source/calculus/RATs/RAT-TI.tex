  \item Find the derivative of $f(x)=\log(x^2+1).$
  \begin{enumerate}
    \item $f'(x)=\frac{1}{x^2+1}$.
    \item $f'(x)=\frac{1}{2x}$.
    \item $f'(x)=\frac{1}{2x}+1$.
    \item $f'(x)=\frac{2x}{x^2+1}$.  %correct
    \item $f'(x)=\log(2x)$.
  \end{enumerate}

  \item Find the derivative of $f(t)=t^2\cdot \cos(t).$
  \begin{enumerate}
    \item $f'(t)=2t\sin(t)$.
    \item $f'(t)=-2t\sin(t)$.
    \item $f'(t)=2t\cos(t)+t^2\sin(t)$.
    \item $f'(t)=2t\cos(t)-t^2\sin(t)$.  %correct
    \item $f'(t)=t^2\cos(t)-2t\sin(t)$.
  \end{enumerate}
  
  \item Evaluate the integral  $\int_{1}^4\frac{1}{x}+xdx.$
  \begin{enumerate}
    \item $\log(x)+\frac{x^2}{2}+C$.
    \item $8+\log(4)$.  
    \item $7.5+\log(4)$. %correct
    \item $7+\log(4)$. 
    \item $7$.
  \end{enumerate}
  
  \item Compute the  antiderivative of $2e^{x}+\sqrt{x}$.
  \begin{enumerate}
    \item $2e^{x}+\frac{1}{2\sqrt{x}}+C$.
    \item $e^{x}+\frac{1}{2\sqrt{x}}+C$.
    \item $2e^{x}+\frac{2}{3}x^{3/2}+C$. %correct
    \item $e^{x}+\frac{2}{3}x^{3/2}+C$. 
    \item $2e^{2x}+\frac{2}{3}x^{3/2}+C$.
  \end{enumerate}
  
  \item Evaluate the following derivative: $\frac{d}{d\theta}[\tan(\theta)]$
  \begin{enumerate}
    \item $\sin^2(\theta)$.
    \item $\cos^2(\theta)$.
    \item $\sec^2(\theta)$. %correct
    \item $\csc^2(\theta)$.
    \item $\cot(\theta)$.
  \end{enumerate}
  
  \item Solve for values of $\theta\in[0, 2\pi)$ such that \[\sin^2(\theta)+\cos(\theta)=1\]
  (note that \(\cos(\theta)\) lacks an exponent).
  \begin{enumerate}
    \item All values of $\theta$ satisfy this equality.
    \item $\theta= 0, \pi/2, \pi$.
    \item $\theta= 0, \pi$.
    \item $\theta= 0, \pi/2, \frac{3\pi}{2}$. %correct
    \item $\theta= \pi/4, \pi/2, \frac{3\pi}{4}$.
  \end{enumerate}
  
  \newpage
  
  \item Which of the following expressions is equal to \[\frac{2}{x}+\frac{1}{x+2}?\]
  \begin{enumerate}
    \item $\frac{3x+4}{x^2+2x}$.  %correct
    \item $\frac{3}{x^2+2x}$.
    \item $\frac{2}{x^2+2x}$.
    \item $\frac{3x+4}{2x+2}$.
    \item $\frac{3}{2x+2}$.
  \end{enumerate}
  
  \item Compute the limit \[\lim_{x\to\infty} \frac{5x^2+3}{3x^3-1}.\]
  \begin{enumerate}
    \item $0$.  %correct
    \item $\infty$.
    \item $\frac{5}{3}$.
    \item $\frac{10}{9}$.
    \item $-3$.
  \end{enumerate}
  
  
  \item Compute the limit \[\lim_{x\to0} \frac{3x^2+2x+4}{4x^2-x+1}.\]
  \begin{enumerate}
    \item $\infty$.
    \item $-\infty$.
    \item $0$.
    \item $\frac{3}{4}$.
    \item $4$.  %correct
  \end{enumerate}


\item Compute the limit \[\lim_{x\to2} \frac{x^2+3x-10}{x^2-4}.\]
  \begin{enumerate}
    \item $0$.
    \item $1$.
    \item $\frac{7}{4}$.  %correct    
    \item $\infty$.
    \item undefined.
  \end{enumerate}
